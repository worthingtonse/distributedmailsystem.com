Here is a simple, effective Python script using the Pillow library. This script takes your "Blank" PNG, selects a specific font, and stamps the DMS address and the date onto the image at exact coordinates.

1. The Python "Stamper" Script
First, you'll need to install the library: pip install Pillow

from PIL import Image, ImageDraw, ImageFont
from datetime import datetime
import uuid

def generate_dms_key_image(template_path, user_address, output_folder):
    # 1. Load the template (e.g., your Mega or Giga PNG)
    img = Image.open(template_path)
    draw = ImageDraw.Draw(img)

    # 2. Define Fonts (You can use a .ttf file for a custom look)
    # Use a monospaced font to make it look like a technical "Deed"
    try:
        font_address = ImageFont.truetype("Courier-Bold.ttf", 40)
        font_date = ImageFont.truetype("Courier.ttf", 25)
    except:
        font_address = ImageFont.load_default()
        font_date = ImageFont.load_default()

    # 3. Define the Content
    mint_date = f"MINTED: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"
    
    # 4. Draw the text onto the image
    # Note: Adjust (x, y) coordinates based on where the "empty space" is in your PNG
    draw.text((100, 750), user_address, fill=(40, 40, 40), font=font_address)
    draw.text((100, 810), mint_date, fill=(80, 80, 80), font=font_date)

    # 5. Save with a unique, non-guessable filename
    filename = f"DMS_Key_{uuid.uuid4().hex}.png"
    save_path = f"{output_folder}/{filename}"
    img.save(save_path)
    
    return filename
